292 @GetMapping("/search")
293 public String searchUsers(@RequestParam(required = false) String username, Model model, RedirectAttributes redirectAttributes) {
294 if (token == null) {
295 return "redirect:/";
296 }
297
298 try {
299 // Si el término de búsqueda es null o vacío, redirigir al feed
300 if (username == null || username.trim().isEmpty()) {
301 return "redirect:/feed";
302 }
303
304 // Obtener todos los usuarios
305 List<User> allUsers = linkAutoServiceProxy.getAllUsers();
306
307 // Filtrar usuarios que coincidan con el término de búsqueda (ignorando mayúsculas y minúsculas)
308 List<User> matchingUsers = allUsers.stream()
309
310 .filter(user -> user.username().toLowerCase().contains(username.toLowerCase()))
311 .collect(Collectors.toList());
312
313 Set <String> verifiedUsers = new HashSet<>();
314 for (User user : matchingUsers) {
315 Boolean isUserVerified = linkAutoServiceProxy.isUserVerified(user.username());
316 if (isUserVerified) {
317 verifiedUsers.add(user.username());
318 }
319 }
320 model.addAttribute("verifiedUsers", verifiedUsers);
321
322 model.addAttribute("searchTerm", username);
323 model.addAttribute("users", matchingUsers);
324
325 // Añadir datos del usuario actual
326 User currentUser = linkAutoServiceProxy.getUserProfile(token);
327 model.addAttribute("currentUser", currentUser);
328 model.addAttribute("username", this.username);
329 model.addAttribute("profilePicture", currentUser.profilePicture());
330 model.addAttribute("role", currentUser.role());
331
332 // Obtener usuarios que sigue el usuario actual
333 List<User> followings = linkAutoServiceProxy.getUserFollowing(this.username);
334 List<String> followingUsernames = followings.stream()
335 .map(User::username)
336 .collect(Collectors.toList());
337 model.addAttribute("followings", followingUsernames);
338
339 return "searchResults"; // Nueva plantilla para mostrar resultados
340 } catch (Exception e) {
341 redirectAttributes.addFlashAttribute("error", "Error al buscar usuarios: " + e.getMessage());
342 return "redirect:/feed";
343 }
