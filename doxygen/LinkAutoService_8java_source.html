179
180 // Recupera el usuario desde DB para que tenga su Set<Post> sincronizado
181 User user = userRepository.findByUsername(u.getUsername()).orElse(null);

183
184 boolean added = user.getSavedPosts().add(post); // Evita duplicados si ya existe en Set
185 if (!added) return false; // Ya estaba guardado

191 @Transactional
192 public Boolean unsavePost (Long postId, User u) {
193 Post post = postRepository.findById(postId).orElse(null);
194 if (post == null) return false;
195
196 User user = userRepository.findByUsername(u.getUsername()).orElse(null);
197 // Buscar el post a eliminar dentro del Set de guardados
198 Post postToRemove = user.getSavedPosts().stream()
199 .filter(p -> p.getId().equals(postId))
200 .findFirst()
201 .orElse(null);
202
203 if (postToRemove != null) {
204 user.getSavedPosts().remove(postToRemove);
205 userRepository.save(user);
206 userRepository.flush();
207 return true;
208 }
